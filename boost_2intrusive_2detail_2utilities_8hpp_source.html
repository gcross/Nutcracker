00435 //This function uses binary search to discover the
00436 //highest set bit of the integer
00437 inline std::size_t floor_log2 (std::size_t x)
00438 {
00439    const std::size_t Bits = sizeof(std::size_t)*CHAR_BIT;
00440    const bool Size_t_Bits_Power_2= !(Bits & (Bits-1));
00441    BOOST_STATIC_ASSERT(Size_t_Bits_Power_2);
00442 
00443    std::size_t n = x;
00444    std::size_t log2 = 0;
00445    
00446    for(std::size_t shift = Bits >> 1; shift; shift >>= 1){
00447       std::size_t tmp = n >> shift;
00448       if (tmp)
00449          log2 += shift, n = tmp;
00450    }
00451 
00452    return log2;
00453 }

00455 inline float fast_log2 (float val)
00456 {
00457    boost::uint32_t * exp_ptr =
00458       static_cast<boost::uint32_t *>(static_cast<void*>(&val));
00459    boost::uint32_t x = *exp_ptr;
00460    const int log_2 = (int)(((x >> 23) & 255) - 128);
00461    x &= ~(255 << 23);
00462    x += 127 << 23;
00463    *exp_ptr = x;
00464 
00465    val = ((-1.0f/3) * val + 2) * val - 2.0f/3;
00466 
00467    return (val + log_2);
00468 }
