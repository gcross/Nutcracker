00041     // Type of a local boost_scope_exit_args variable.
00042     // First use in a local scope will declare the boost_scope_exit_args
00043     // variable, subsequent uses will be resolved as two comparisons
00044     // (cmp1 with 0 and cmp2 with boost_scope_exit_args).
00045     template<int Dummy = 0>
00046     struct declared
00047     {
00048         void* value;
00049         static int const cmp2 = 0;
00050         friend void operator>(int, declared const&) {}
00051     };
00052 
00053     struct undeclared { declared<> dummy[2]; };

00055     template<int> struct resolve;
00056 
00057     template<>
00058     struct resolve<sizeof(declared<>)>
00059     {
00060         static const int cmp1 = 0;
00061     };

00063     template<>
00064     struct resolve<sizeof(undeclared)>
00065     {
00066         template<int>
00067         struct cmp1
00068         {
00069             static int const cmp2 = 0;
00070         };
00071     };

00164     // to determine the type of 'this' due to error C2355 being
00165     // incorrectly reported (a compiler bug that's present in 8 and 9

00170     // 
00171     // The implementation below is the more general, type-index
00172     // sizeof()-based approach implemented in Boost.Typeof which
00173     // appears to work fine for typeof(this) provided the reference to
00174     // 'this' is not present in the template argument list (i.e. the
00175     // type-index is captured into a local enumerator before being
00176     // used as a template argument).

00251 // Expand to 1 if the first sequence element is 'this'
00252 // or 0 otherwise.
00253 //
00254 // Mechanics: [given that 'this(x)' --> '(this(x))']
00255 //
00256 //   head = front(seq)  --> 'this'      or 'somearg'
00257 //   funcexp = head(_)  --> '(this(_))' or 'somearg(_)'
00258 //   size_and_possible_tokens = seq_size(funcexp)
00259 //   yields either
00260 //        0 somearg(_)
00261 //   or   1
00262 //   expansion of 'cat(X,size_and_possible_tokens _) )'
00263 //   yields either
00264 //        X0 _ )        --> zero( _ )       --> 0
00265 //   or   X1 tokens _ ) --> one( tokens _ ) --> 1
00266 //
00267 // where X0     is BOOST_SCOPE_EXIT_HAS_THIS_CONSUME_0
00268 //       X1     is BOOST_SCOPE_EXIT_HAS_THIS_CONSUME_1
00269 //       zero() is BOOST_SCOPE_EXIT_HAS_THIS_FALSE()
00270 //       one()  is BOOST_SCOPE_EXIT_HAS_THIS_TRUE()
00271 //
00272 #define BOOST_SCOPE_EXIT_HAS_THIS(seq)                            \
00273    BOOST_PP_CAT(BOOST_SCOPE_EXIT_HAS_THIS_CONSUME_,               \
00274                 BOOST_PP_SEQ_SIZE(BOOST_PP_SEQ_HEAD(seq)(_)) _) ) \
00275 
00276 #define BOOST_SCOPE_EXIT_HAS_THIS_CONSUME_0 BOOST_SCOPE_EXIT_HAS_THIS_FALSE(
00277 #define BOOST_SCOPE_EXIT_HAS_THIS_CONSUME_1 BOOST_SCOPE_EXIT_HAS_THIS_TRUE(
00278 
00279 #define BOOST_SCOPE_EXIT_HAS_THIS_FALSE(x) 0
00280 #define BOOST_SCOPE_EXIT_HAS_THIS_TRUE(x)  1
